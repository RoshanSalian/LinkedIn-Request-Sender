// Find all buttons on the page with a span child element containing "JOIN" or "REJOIN"
const buttons = document.querySelectorAll('button > span:contains("JOIN"), button > span:contains("REJOIN")');

// Loop through each button
for (let i = 0; i < buttons.length; i++) {
  const button = buttons[i].parentNode; // get the parent button element

  // Check if the child span element contains the text "JOIN" or "REJOIN"
  if (buttons[i].textContent === 'JOIN') {
    // Add an event listener to the button
    button.addEventListener('click', function() {
      // Trigger an action when the button is clicked
      console.log('JOIN button clicked!');
    });
  } else if (buttons[i].textContent === 'REJOIN') {
    // Add an event listener to the button
    button.addEventListener('click', function() {
      // Trigger an action when the button is clicked
      console.log('REJOIN button clicked!');
    });
  }
}
